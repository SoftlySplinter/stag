Stag

A simple language which compiles to Java bytecode.

Syntax

state HttpRequest {
  URL url;

  transition(HttpResponse) {
    HttpUrlConnection conn = (HttpUrlConnection) url.openConnection();
    return new HttpResponse(conn);
  }
}
